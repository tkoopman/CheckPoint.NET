Add new group with exclusion using GroupWithExclusion(Session, Boolean)
Copy
var gwe = new GroupWithExclusion(Session) {
    Name = "MyGroupWithExclusion",
    Include = ObjectSummary.Any,
    Except = Session.FindGroup("ExcludeGroup")
};
gwe.AcceptChanges();
